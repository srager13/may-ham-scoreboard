# Frontend development
cd frontend
npm run dev          # Development server with hot reload
npm run build        # Build for production

# Full-stack development  
go run .             # Serves both API and React app

Usage Example:

```
import MatchScreen from './components/MatchScreen';
import { golfAPI } from './services/api';

// In your app, after login:
golfAPI.setToken(userToken);

// Then render the match screen:
<MatchScreen matchId="uuid-of-match" />
```

-------------------------------------------------------------
Leaderboard
-------------------------------------------------------------
Data structure expected for the Leaderboard
{
  tournament: { id, name, status, start_date, end_date },
  team_standings: [
    {
      team: { id, name, color },
      points_won, points_lost,
      matches_won, matches_lost, matches_tied,
      holes_won, holes_lost, holes_tied
    }
  ],
  individual_standings: [
    {
      user: { id, name, handicap },
      team: { id, name, color },
      points_won, points_lost,
      matches_played, matches_won, matches_lost, matches_tied,
      holes_won, holes_lost, holes_tied,
      win_percentage
    }
  ],
  live_matches: [
    {
      id, match_number, format,
      current_hole, total_holes,
      team1, team2,
      team1_up, // positive = team1 leading, negative = team2 leading
      status, last_update
    }
  ]
}
